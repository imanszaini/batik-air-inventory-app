    <!-- Search -->
    <h2>Search by PN or SN</h2>
    <div class="search-box">
      <input type="text" id="searchInput" placeholder="Enter PN or SN to track">
      <button onclick="searchItem()">Search</button>
    </div>

    <!-- Results -->
    <table id="inventoryTable">
      <thead>
        <tr>
          <th>PN Description</th>

  <script>
    let items = JSON.parse(localStorage.getItem('inventoryItems')) || [];

    const form = document.getElementById('addForm');
    const tableBody = document.querySelector('#inventoryTable tbody');

    // Render full inventory
    function renderTable(highlightIndex = null) {
      tableBody.innerHTML = '';
      items.forEach((item, index) => {
        const row = document.createElement('tr');
        if (highlightIndex === index) row.classList.add('highlight');
        row.innerHTML = `
          <td>${item.pn}</td>
          <td>${item.sn}</td>
          <td>${item.batch}</td>
          <td>${item.bin}</td>
          <td>${item.location}</td>
        `;
        tableBody.appendChild(row);
      });
    }

    // Add item
    form.addEventListener('submit', e => {
      e.preventDefault();
      const newItem = {
        pn: document.getElementById('pn').value.trim(),
        sn: document.getElementById('sn').value.trim(),
        batch: document.getElementById('batch').value.trim(),
        bin: document.getElementById('bin').value.trim(),
        location: document.getElementById('location').value
      };
      items.push(newItem);
      localStorage.setItem('inventoryItems', JSON.stringify(items));
      renderTable();
      form.reset();
    });

    // Search function
    function searchItem() {
      const query = document.getElementById('searchInput').value.trim().toLowerCase();
      if (!query) {
        renderTable();
        return;
      }
      const foundIndex = items.findIndex(item =>
        item.pn.toLowerCase() === query || item.sn.toLowerCase() === query
      );
      if (foundIndex >= 0) {
        renderTable(foundIndex); // highlight found row
      } else {
        alert("Item not found!");
      }
    }

    // Initial render
    renderTable();
  </script>
